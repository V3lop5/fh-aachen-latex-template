% Add Glossary Entries here
\newglossaryentry{framework}
{
  name=Framework,
  description={bezeichnet eine Grundstruktur bestehend aus Bibliotheken, Komponenten und Strukturen für die Entwicklung einer spezifischen Anwendung. Es ist kein eigenständiges Programm, sondern gibt nur einen Rahmen für die Implementierung vor. Ziel eines Frameworks ist die Vereinfachung der Implementierung von Anwendungsprogrammen.}
}
